4 #permalink Zdjęcie dnia - sekcja "Zdjęcia dnia" strony

Toggle example guides Toggle HTML markup

To jest strona szablonu ze zdjęciem dnia pobieranym od NASA.

#content: Div o tym id ma padding 40px, szerokość 917px i wyjustowany tekst w środku.

.bigtitle: Elementy o tej klasie mają pogrubioną czcionkę o rozmiarze 32px, litery są wystrzelone na 2px.

.dottedline: Elementy o tej klasie mają wysokość 5px, margines górny 15px, margines dolny 20px i dole obramowanie 2px.

#nasa-apod: Div o tym id umożliwia wczytanie zdjęcia za pomocą API ze strony NASA.

<div id="content">

		</div>
	</div>
<script>
    async function fetchAPOD() {
        try {
            const apiKey = 'WY1QbfQACxdGXRbDJflTWlNBmsZGzpgXuok1naaX';
            const response = await fetch(`https://api.nasa.gov/planetary/apod?api_key=${apiKey}`);
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            const data = await response.json();
            displayAPOD(data);
        } catch (error) {
            console.error('Could not fetch APOD data:', error);
        }
    }
    function displayAPOD(data) {
        const imageElement = document.getElementById('apod-image');
        imageElement.src = data.url;
        imageElement.alt = data.title;
    }
    document.addEventListener('DOMContentLoaded', fetchAPOD);
</script>
